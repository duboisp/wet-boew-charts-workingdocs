Donut preset wb-charts-donut

Donut Preset JSON Source Code 

// FYI - The "Donut" preset is already defined by default, this are just to show an example on you may reproduce it in your own preset.
var wet_boew_charts = {
	flot: {
		donut: {
			series: {
				pie: {
					show: true,
					radius: 1,
					label: {
						show: true,
						radius: 1,
						threshold: 0.08
					},
					tilt: 0.5,
					innerRadius: 0.45,
					startAngle: 1
				}
			},
			grid: {
				hoverable: true
			}
		},
	charts: {
		donut: {
			decimal: 1
		}
	}
}
